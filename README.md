## 설치 명령어 (Environment Setup)
python -m venv venv

.\venv\Scripts\activate

pip install flask

python -m pip install --upgrade pip

pip install flask-migrate

pip install flask-wtf

set FLASK_APP=pybo

set FLASK_DEBUG=true

flask db init

flask db migrate 

flask db upgrade 

flask run
